apiVersion: v1
kind: Service
metadata:
  name: my-clusterip-service
spec:
  selector:
    app: myapp  # Match the Pods with this label
  ports:
    - protocol: TCP
      port: 80  # Port exposed by the service
      targetPort: 80  # Port on the Pod
  type: ClusterIP  # Exposes the service within the cluster (default)
